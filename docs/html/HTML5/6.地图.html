function TMap(key) {//这个key是在腾讯地图申请的key
	return new Promise(function(resolve, reject) {
		//在初始化的时候将成功状态下的数据以赋给形参 ‘qq’，用then方法接收这个形参
		window.init = function() {
			resolve(qq) //注意这里
		}
		//动态创建script标签，嵌入head标签内
		var script = document.createElement("script");
		script.type = "text/javascript";
		script.src = "http://map.qq.com/api/js?v=2.exp&callback=init&key=" + key;
	  		script.onerror = reject;
	  		document.head.appendChild(script);
	  	})
}
function loadmark(taskArr,ImgSrc,map) {////封装画点的函数，形参是后台传过来的这个半径内一共有多少个站点的参数,以及图标的路径ImgSrc,和地图容器map
	for(var i = 0; i < taskArr.length; i++) {
		let latLng = new qq.maps.LatLng(taskArr[i].latitude, taskArr[i].longitude);//将站点的经纬度赋值
		let marker = new qq.maps.Marker({//将定位定在赋值的经纬度上，
			position: latLng,//定位
			map: map//显示在哪个容器里
		});
		let anchor = new qq.maps.Point(0, 39);//图标下方为一个点
		let	size = new qq.maps.Size(42, 68);//设置图标尺寸
		let	origin = new qq.maps.Point(0, 0);//origin是切图坐标
		let mapimg = this.mapimg;//图标的路径，在data里有
		let	icon = new qq.maps.MarkerImage(//设置Marker自定义图标属性
			ImgSrc,//图片的路径
			size,//图片的大小
			origin,//切图的坐标
			anchor//图标下方为一个点
		);
		marker.setIcon(icon);//给marker设置图标
		addMyFn(marker, taskArr[i], latLng);//调用已经封装好的函数
	}
}
let StoreName = document.getElementById('StoreName');//获取移上图标时要显示站点信息的节点
let Timer = null;//自动消失消息窗口，设置一个倒计时
function addMyFn(marker, oneData, postion) {//三个参数分别是设置好的图标以及图标的位置尺寸，每个图标对应的那组数据，他们的经纬度
	qq.maps.event.addListener(marker, 'dblclick', function() {//添加点击事件，点击的时候，跳转页面，页面有ID传过去，会获取该站点的那组数据
		window.open('/Home/Store/details/user_id/' + oneData.user_id);
	});
	qq.maps.event.addListener(marker, 'mouseover', function(event) {//添加鼠标以上事件，移上的时候，显示的时候在图标的上方显示站点名称
		StoreName.innerHTML = oneData.userName; //站点的名称，
		StoreName.style.display = 'block'; //是否显示站点的信息，可以用class类的形式控制
		showName(2);
	});
}
function showName(type = 1) {//调用此方法要传入的参数
	clearTimeout(Timer);//进这个函数之前先清楚之前的倒计时
	if(type == 1) {
		StoreName.style.display = 'none';//初始化的时候不会显示，所以直接给display:none;，也可以以class类的形式
		return;
	}
	Timer = setTimeout(function() {
		StoreName.style.display = 'none';//初始化的时候不会显示，所以直接给display:none;，也可以以class类的形式
	}, 5000);
}

<script>
export default{
	import { TMap,loadmark,hideShowStoreName } from '@/utils/map.js'
	data(){
		return{
			customer:'深圳福田国通大厦',//设置默认地址
			far_away:8,//设置默认查询半径
			mapimg:'www.baidu.com',//icon图标的路径
			seversList: [],//站点信息数组
		}
	}
	methods:{
		shutdown(){
			this.customer = '';//当聚焦输入框时，输入框置空
		},
		setServeList(){
			TMap('CIDBZ-PVU6W-D4ZRO-OEYEZ-4TQ76-NPF5F').then(qq => {//前面那一串是腾讯地图申请的key
					if(!this.far_away>=1&&this.far_away<=15){//规定搜索半径
						this.$message({
							message: '搜索半径为1到15公里',
							type: 'warning'
						});
						this.far_away = 8;
						return;
					}
					this.$http.post(this.$apis.serversList, {
						key_words: this.customer,//搜索地址
						far_away: this.far_away//搜索半径
					}).then((resp) => {
					this.seversList = resp.data.serach_list;//站点信息数组
					let r = this.far_away*1000;//搜索半径
					let center = new qq.maps.LatLng(resp.data.lat, resp.data.long)//设置中心点，中心的是客户的位置,参数是经纬度
					let zoomFirst = 12;
					let map = new qq.maps.Map(document.getElementById("container"), {//创建一个地图容器，并且设置中心坐标，上面的center已经有了经纬度，设置地图的精确度zoom
						center: center,// 地图的中心地理坐标。
						zoom: zoomFirst//地图 精确度
					});
					let circle = new qq.maps.Circle({//在地图容器上画圆，
						center: center, //圆心位置
						radius: r, //半径
						zIndex: 1000,//覆盖厚度
						visible: true,//是否可见
					})
					circle.setMap(map); //设置圆圈在哪个容器显示
					loadmark(this.seversList,this.mapimg,map);//画点
					qq.maps.event.addListener(map,'zoom_changed',function(){//map(容地图器)，zoom_changed(地图缩放时触发)，第三个参数是所要执行的代码
							let zoomLevel = map.getZoom();//zoomFirst设置的时候是设置是固定的，zoom是可以动态变化的，当地图缩小时，zoom会变小，当地图变大时，zoom会变大
							if(zoomLevel < zoomFirst) {
								showName(1);//如果地图缩小，zoom变小，则调用hideShowStoreName方法，传形参1过去，让站点信息立即消失
							}
					});
					
				})
			});
		}
	}
	mount(){
		//页面加载完成的时候调用
		this.setServeList()
	}
}
</script>
